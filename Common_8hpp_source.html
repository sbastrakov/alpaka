21 //! All functions that can be used on an accelerator have to be attributed with ALPAKA_FN_ACC or ALPAKA_FN_HOST_ACC.
22 //!
23 //! Usage:
24 //! ALPAKA_FN_ACC
25 //! auto add(std::int32_t a, std::int32_t b)
26 //! -> std::int32_t;
27 #if BOOST_LANG_CUDA || BOOST_LANG_HIP
28  #if defined(ALPAKA_ACC_GPU_CUDA_ONLY_MODE) || defined(ALPAKA_ACC_GPU_HIP_ONLY_MODE)
29  #define ALPAKA_FN_ACC __device__
30  #else
31  #define ALPAKA_FN_ACC __device__ __host__
32  #endif
33  #define ALPAKA_FN_HOST_ACC __device__ __host__
34  #define ALPAKA_FN_HOST __host__
35 #else
36  #define ALPAKA_FN_ACC
37  #define ALPAKA_FN_HOST_ACC
38  #define ALPAKA_FN_HOST
39 #endif

69 //-----------------------------------------------------------------------------
70 //! This macro defines a variable lying in global accelerator device memory.
71 //!
72 //! Example:
73 //! ALPAKA_STATIC_ACC_MEM_GLOBAL int i;
74 //!
75 //! Those variables behave like ordinary variables when used in file-scope.
76 //! They have external linkage (are accessible from other compilation units).
77 //! If you want to access it from a different compilation unit, you have to declare it as extern:
78 //! extern ALPAKA_STATIC_ACC_MEM_GLOBAL int i;
79 //! Like ordinary variables, only one definition is allowed (ODR)
80 //! Failure to do so might lead to linker errors.
81 //!
82 //! In contrast to ordinary variables, you can not define such variables
83 //! as static compilation unit local variables with internal linkage
84 //! because this is forbidden by CUDA.
85 #if (BOOST_LANG_CUDA && BOOST_ARCH_PTX) || (BOOST_LANG_HIP && (BOOST_ARCH_HSA || BOOST_ARCH_PTX))
86  #define ALPAKA_STATIC_ACC_MEM_GLOBAL __device__
87 #else
88  #define ALPAKA_STATIC_ACC_MEM_GLOBAL
89 #endif
90
