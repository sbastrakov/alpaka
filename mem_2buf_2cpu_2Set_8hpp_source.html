40  //! The CPU device ND memory set task base.
41  template<
42  typename TDim,
43  typename TView,
44  typename TExtent>
45  struct TaskSetCpuBase
46  {
47  using ExtentSize = idx::Idx<TExtent>;
48  using DstSize = idx::Idx<TView>;
49  using Elem = elem::Elem<TView>;
50 
51  static_assert(
52  !std::is_const<TView>::value,
53  "The destination view can not be const!");
54 
55  static_assert(
56  dim::Dim<TView>::value == dim::Dim<TExtent>::value,
57  "The destination view and the extent are required to have the same dimensionality!");
58  static_assert(
59  dim::Dim<TView>::value == TDim::value,
60  "The destination view and the input TDim are required to have the same dimensionality!");
61 
62  static_assert(
63  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
64  "The view and the extent are required to have compatible idx type!");
65 
66  //-----------------------------------------------------------------------------
67  TaskSetCpuBase(
68  TView & view,
69  std::uint8_t const & byte,
70  TExtent const & extent) :
71  m_byte(byte),
72  m_extent(extent::getExtentVec(extent)),
73  m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem))),
74 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
75  m_dstExtent(extent::getExtentVec(view)),
76 #endif
77  m_dstPitchBytes(mem::view::getPitchBytesVec(view)),
78  m_dstMemNative(reinterpret_cast<std::uint8_t *>(mem::view::getPtrNative(view)))
79  {
80  ALPAKA_ASSERT((vec::cast<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
81  ALPAKA_ASSERT(m_extentWidthBytes <= m_dstPitchBytes[TDim::value - 1u]);
82  }
83 
84 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL

96  }
97 #endif

110  //! The CPU device ND memory set task.
111  template<
112  typename TDim,
113  typename TView,
114  typename TExtent>
115  struct TaskSetCpu : public TaskSetCpuBase<TDim, TView, TExtent>
116  {
117  using DimMin1 = dim::DimInt<TDim::value - 1u>;
118  using typename TaskSetCpuBase<TDim, TView, TExtent>::ExtentSize;
119  using typename TaskSetCpuBase<TDim, TView, TExtent>::DstSize;
120 
121  //-----------------------------------------------------------------------------
122  using TaskSetCpuBase<TDim, TView, TExtent>::TaskSetCpuBase;
123 
124  //-----------------------------------------------------------------------------
125  ALPAKA_FN_HOST auto operator()() const
126  -> void
127  {
128  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
129 
130 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
131  this->printDebug();
132 #endif
133  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
134  vec::Vec<DimMin1, ExtentSize> const extentWithoutInnermost(vec::subVecBegin<DimMin1>(this->m_extent));
135  // [z, y, x] -> [y, x] because the z pitch (the full idx of the buffer) is not required.
136  vec::Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(vec::subVecEnd<DimMin1>(this->m_dstPitchBytes));
137 
138  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
139  {
140  meta::ndLoopIncIdx(
141  extentWithoutInnermost,
142  [&](vec::Vec<DimMin1, ExtentSize> const & idx)
143  {
144 
145  memset(
146  reinterpret_cast<void *>(this->m_dstMemNative + (vec::cast<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
147  this->m_byte,
148  static_cast<std::size_t>(this->m_extentWidthBytes));
149  });
150  }
151  }
152  };
