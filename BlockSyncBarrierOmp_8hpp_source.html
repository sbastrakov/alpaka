120  block::sync::BlockSyncBarrierOmp const & blockSync,
121  int predicate)
122  -> int
123  {
124  // The first thread initializes the value.
125  // There is an implicit barrier at the end of omp single.
126  // NOTE: This code is executed only once for all OpenMP threads.
127  // If multiple blocks with multiple threads are executed in parallel
128  // this reduction is executed only for one block!
129  #pragma omp single
130  {
131  ++blockSync.m_generation;
132  blockSync.m_result[blockSync.m_generation % 2u] = TOp::InitialValue;
133  }
134 
135  auto const generationMod2(blockSync.m_generation % 2u);
136  int& result(blockSync.m_result[generationMod2]);
137  bool const predicateBool(predicate != 0);
138 
139  detail::AtomicOp<TOp>()(result, predicateBool);
140 
141  // Wait for all threads to write their predicate into the vector.
142  // NOTE: This waits for all threads in all blocks.
143  // If multiple blocks are executed in parallel this is not optimal.
144  #pragma omp barrier
145 
146  return blockSync.m_result[generationMod2];
147  }
