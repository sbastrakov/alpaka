40  //! The CPU device event implementation.
41  template<
42  typename TDev>
43  class EventGenericThreadsImpl final : public concepts::Implements<wait::ConceptCurrentThreadWaitFor, EventGenericThreadsImpl<TDev>>
44  {
45  public:
46  //-----------------------------------------------------------------------------
47  EventGenericThreadsImpl(
48  TDev const & dev) noexcept :
49  m_dev(dev),
50  m_mutex(),
51  m_enqueueCount(0u),
52  m_LastReadyEnqueueCount(0u)
53  {}
54  //-----------------------------------------------------------------------------
55  EventGenericThreadsImpl(EventGenericThreadsImpl<TDev> const &) = delete;
56  //-----------------------------------------------------------------------------
57  EventGenericThreadsImpl(EventGenericThreadsImpl<TDev> &&) = delete;
58  //-----------------------------------------------------------------------------
59  auto operator=(EventGenericThreadsImpl<TDev> const &) -> EventGenericThreadsImpl<TDev> & = delete;
60  //-----------------------------------------------------------------------------
61  auto operator=(EventGenericThreadsImpl<TDev> &&) -> EventGenericThreadsImpl<TDev> & = delete;
62  //-----------------------------------------------------------------------------
63  ~EventGenericThreadsImpl() noexcept = default;
64 
65  //-----------------------------------------------------------------------------
66  auto isReady() noexcept -> bool
67  {
68  return (m_LastReadyEnqueueCount == m_enqueueCount);
69  }
70 
71  //-----------------------------------------------------------------------------
72  auto wait(std::size_t const & enqueueCount, std::unique_lock<std::mutex>& lk) const noexcept -> void
73  {
74  ALPAKA_ASSERT(enqueueCount <= m_enqueueCount);
75 
76  while(enqueueCount > m_LastReadyEnqueueCount)
77  {
78  auto future = m_future;
79  lk.unlock();
80  future.get();
81  lk.lock();
82  }
83  }
84 
85  public:
86  TDev const m_dev; //!< The device this event is bound to.
87 
88  std::mutex mutable m_mutex; //!< The mutex used to synchronize access to the event.
89  std::shared_future<void> m_future; //!< The future signaling the event completion.
90  std::size_t m_enqueueCount; //!< The number of times this event has been enqueued.
91  std::size_t m_LastReadyEnqueueCount; //!< The time this event has been ready the last time.
92  //!< Ready means that the event was not waiting within a queue (not enqueued or already completed).
93  //!< If m_enqueueCount == m_LastReadyEnqueueCount, the event is currently not enqueued
94  };
95  }
96  }

202  event::EventGenericThreads<TDev> & event)
203  -> void
204  {
205 #if (BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
206  alpaka::ignore_unused(queueImpl);
207 #endif
208  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
209 
210  // Copy the shared pointer of the event implementation.
211  // This is forwarded to the lambda that is enqueued into the queue to ensure that the event implementation is alive as long as it is enqueued.
212  auto spEventImpl(event.m_spEventImpl);
213 
214  // Setting the event state and enqueuing it has to be atomic.
215  std::lock_guard<std::mutex> lk(spEventImpl->m_mutex);
216 
217  ++spEventImpl->m_enqueueCount;
218 
219 // Workaround: Clang can not support this when natively compiling device code. See ConcurrentExecPool.hpp.
220 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
221  auto const enqueueCount = spEventImpl->m_enqueueCount;
222 
223  // Enqueue a task that only resets the events flag if it is completed.
224  spEventImpl->m_future = queueImpl.m_workerThread.enqueueTask(
225  [spEventImpl, enqueueCount]()
226  {
227  std::unique_lock<std::mutex> lk2(spEventImpl->m_mutex);
228 
229  // Nothing to do if it has been re-enqueued to a later position in the queue.
230  if(enqueueCount == spEventImpl->m_enqueueCount)
231  {
232  spEventImpl->m_LastReadyEnqueueCount = spEventImpl->m_enqueueCount;
233  }
234  });
235 #endif
236  }
237  };

324  template<typename TDev>
325  ALPAKA_FN_HOST auto currentThreadWaitForDevice(
326  TDev const & dev
327  )
328  ->void
329  {
330  // Get all the queues on the device at the time of invocation.
331  // All queues added afterwards are ignored.
332  auto vQueues(dev.getAllQueues());
333  // Furthermore there should not even be a chance to enqueue something between getting the queues and adding our wait events!
334  std::vector<event::EventGenericThreads<TDev>> vEvents;
335  for(auto && spQueue : vQueues)
336  {
337  vEvents.emplace_back(dev);
338  spQueue->enqueue(vEvents.back());
339  }
340 
341  // Now wait for all the events.
342  for(auto && event : vEvents)
343  {
344  wait::wait(event);
345  }
346  }
347  }
