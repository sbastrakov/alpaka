388  template<
389  typename TDim,
390  typename TView>
391  struct Print
392  {
393  ALPAKA_FN_HOST static auto print(
394  TView const & view,
395  elem::Elem<TView> const * const ptr,
396  vec::Vec<dim::Dim<TView>, idx::Idx<TView>> const & extent,
397  std::ostream & os,
398  std::string const & elementSeparator,
399  std::string const & rowSeparator,
400  std::string const & rowPrefix,
401  std::string const & rowSuffix)
402  -> void
403  {
404  os << rowPrefix;
405 
406  auto const pitch(view::getPitchBytes<TDim::value+1u>(view));
407  auto const lastIdx(extent[TDim::value]-1u);
408  for(auto i(decltype(lastIdx)(0)); i<=lastIdx ;++i)
409  {
410  Print<
411  dim::DimInt<TDim::value+1u>,
412  TView>
413  ::print(
414  view,
415  reinterpret_cast<elem::Elem<TView> const *>(reinterpret_cast<std::uint8_t const *>(ptr)+i*pitch),
416  extent,
417  os,
418  elementSeparator,
419  rowSeparator,
420  rowPrefix,
421  rowSuffix);
422 
423  // While we are not at the end of a row, add the row separator.
424  if(i != lastIdx)
425  {
426  os << rowSeparator;
427  }
428  }
429 
430  os << rowSuffix;
431  }
432  };
433  //-----------------------------------------------------------------------------
434  template<
435  typename TView>
436  struct Print<
437  dim::DimInt<dim::Dim<TView>::value-1u>,
438  TView>
439  {
440  ALPAKA_FN_HOST static auto print(
441  TView const & view,
442  elem::Elem<TView> const * const ptr,
443  vec::Vec<dim::Dim<TView>, idx::Idx<TView>> const & extent,
444  std::ostream & os,
445  std::string const & elementSeparator,
446  std::string const & rowSeparator,
447  std::string const & rowPrefix,
448  std::string const & rowSuffix)
449  -> void
450  {
451  alpaka::ignore_unused(view);
452  alpaka::ignore_unused(rowSeparator);
453 
454  os << rowPrefix;
455 
456  auto const lastIdx(extent[dim::Dim<TView>::value-1u]-1u);
457  for(auto i(decltype(lastIdx)(0)); i<=lastIdx ;++i)
458  {
459  // Add the current element.
460  os << *(ptr+i);
461 
462  // While we are not at the end of a line, add the element separator.
463  if(i != lastIdx)
464  {
465  os << elementSeparator;
466  }
467  }
468 
469  os << rowSuffix;
470  }
471  };
