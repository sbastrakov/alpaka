49  //! The CPU OpenMP 4.0 accelerator execution task.
50  template<
51  typename TDim,
52  typename TIdx,
53  typename TKernelFnObj,
54  typename... TArgs>
55  class TaskKernelCpuOmp4 final :
56  public workdiv::WorkDivMembers<TDim, TIdx>
57  {
58  public:
59  //-----------------------------------------------------------------------------
60  template<
61  typename TWorkDiv>
62  ALPAKA_FN_HOST TaskKernelCpuOmp4(
63  TWorkDiv && workDiv,
64  TKernelFnObj const & kernelFnObj,
65  TArgs && ... args) :
66  workdiv::WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv)),
67  m_kernelFnObj(kernelFnObj),
68  m_args(std::forward<TArgs>(args)...)
69  {
70  static_assert(
71  dim::Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
72  "The work division and the execution task have to be of the same dimensionality!");
73  }
74  //-----------------------------------------------------------------------------
75  TaskKernelCpuOmp4(TaskKernelCpuOmp4 const & other) = default;
76  //-----------------------------------------------------------------------------
77  TaskKernelCpuOmp4(TaskKernelCpuOmp4 && other) = default;
78  //-----------------------------------------------------------------------------
79  auto operator=(TaskKernelCpuOmp4 const &) -> TaskKernelCpuOmp4 & = default;
80  //-----------------------------------------------------------------------------
81  auto operator=(TaskKernelCpuOmp4 &&) -> TaskKernelCpuOmp4 & = default;
82  //-----------------------------------------------------------------------------
83  ~TaskKernelCpuOmp4() = default;
84 
85  //-----------------------------------------------------------------------------
86  //! Executes the kernel function object.
87  ALPAKA_FN_HOST auto operator()() const
88  -> void
89  {
90  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
91 
92  auto const gridBlockExtent(
93  workdiv::getWorkDiv<Grid, Blocks>(*this));
94  auto const blockThreadExtent(
95  workdiv::getWorkDiv<Block, Threads>(*this));
96  auto const threadElemExtent(
97  workdiv::getWorkDiv<Thread, Elems>(*this));
98 
99  // Get the size of the block shared dynamic memory.
100  auto const blockSharedMemDynSizeBytes(
101  meta::apply(
102  [&](std::decay_t<TArgs> const & ... args)
103  {
104  return
105  kernel::getBlockSharedMemDynSizeBytes<
106  acc::AccCpuOmp4<TDim, TIdx>>(
107  m_kernelFnObj,
108  blockThreadExtent,
109  threadElemExtent,
110  args...);
111  },
112  m_args));
113 
114 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
115  std::cout << __func__
116  << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B" << std::endl;
117 #endif
118  // Bind all arguments except the accelerator.
119  // TODO: With C++14 we could create a perfectly argument forwarding function object within the constructor.
120  auto const boundKernelFnObj(
121  meta::apply(
122  [this](std::decay_t<TArgs> const & ... args)
123  {
124  return
125  std::bind(
126  std::ref(m_kernelFnObj),
127  std::placeholders::_1,
128  std::ref(args)...);
129  },
130  m_args));
131 
132  // The number of blocks in the grid.
133  TIdx const gridBlockCount(gridBlockExtent.prod());
134  // The number of threads in a block.
135  TIdx const blockThreadCount(blockThreadExtent.prod());
136 
137  // We have to make sure, that the OpenMP runtime keeps enough threads for executing a block in parallel.
138  auto const maxOmpThreadCount(::omp_get_max_threads());
139  auto const maxTeamCount(maxOmpThreadCount/static_cast<int>(blockThreadCount));
140  auto const teamCount(std::min(maxTeamCount, static_cast<int>(gridBlockCount)));
141 
142  if(::omp_in_parallel() != 0)
143  {
144  throw std::runtime_error("The OpenMP 4.0 backend can not be used within an existing parallel region!");
145  }
146 
147  // Force the environment to use the given number of threads.
148  int const ompIsDynamic(::omp_get_dynamic());
149  ::omp_set_dynamic(0);
150 
151  // `When an if(scalar-expression) evaluates to false, the structured block is executed on the host.`
152  #pragma omp target if(0)
153  {
154  #pragma omp teams num_teams(teamCount) thread_limit(blockThreadCount)
155  {
156 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
157  // The first team does some checks ...
158  if((::omp_get_team_num() == 0))
159  {
160  int const iNumTeams(::omp_get_num_teams());
161  printf("%s omp_get_num_teams: %d\n", __func__, iNumTeams);
162  }
163 #endif
164  acc::AccCpuOmp4<TDim, TIdx> acc(
165  *static_cast<workdiv::WorkDivMembers<TDim, TIdx> const *>(this),
166  blockSharedMemDynSizeBytes);
167 
168  #pragma omp distribute
169  for(TIdx b = 0u; b<gridBlockCount; ++b)
170  {
171  vec::Vec<dim::DimInt<1u>, TIdx> const gridBlockIdx(b);
172  // When this is not repeated here:
173  // error: gridBlockExtent referenced in target region does not have a mappable type
174  auto const gridBlockExtent2(
175  workdiv::getWorkDiv<Grid, Blocks>(*static_cast<workdiv::WorkDivMembers<TDim, TIdx> const *>(this)));
176  acc.m_gridBlockIdx = idx::mapIdx<TDim::value>(
177  gridBlockIdx,
178  gridBlockExtent2);
179 
180  // Execute the threads in parallel.
181 
182  // Parallel execution of the threads in a block is required because when syncBlockThreads is called all of them have to be done with their work up to this line.
183  // So we have to spawn one OS thread per thread in a block.
184  // 'omp for' is not useful because it is meant for cases where multiple iterations are executed by one thread but in our case a 1:1 mapping is required.
185  // Therefore we use 'omp parallel' with the specified number of threads in a block.
186  #pragma omp parallel num_threads(blockThreadCount)
187  {
188 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
189  // The first thread does some checks in the first block executed.
190  if((::omp_get_thread_num() == 0) && (b == 0))
191  {
192  int const numThreads(::omp_get_num_threads());
193  printf("%s omp_get_num_threads: %d\n", __func__, numThreads);
194  if(numThreads != static_cast<int>(blockThreadCount))
195  {
196  throw std::runtime_error("ERROR: The OpenMP runtime did not use the number of threads that had been required!");
197  }
198  }
199 #endif
200  boundKernelFnObj(
201  acc);
202 
203  // Wait for all threads to finish before deleting the shared memory.
204  // This is done by default if the omp 'nowait' clause is missing
205  //block::sync::syncBlockThreads(acc);
206  }
207 
208  // After a block has been processed, the shared memory has to be deleted.
209  block::shared::st::freeMem(acc);
210  }
211  }
212  }
213 
214  // Reset the dynamic thread number setting.
215  ::omp_set_dynamic(ompIsDynamic);
216  }
217 
218  private:
219  TKernelFnObj m_kernelFnObj;
220  std::tuple<std::decay_t<TArgs>...> m_args;
221  };
222  }
