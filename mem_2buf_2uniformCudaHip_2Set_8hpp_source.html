122  typename TQueue
123  >
124  auto enqueue(TQueue & queue) const
125  -> void
126  {
127  static_assert(
128  dim::Dim<TView>::value == 1u,
129  "The destination buffer is required to be 1-dimensional for this specialization!");
130  static_assert(
131  dim::Dim<TView>::value == dim::Dim<TExtent>::value,
132  "The destination buffer and the extent are required to have the same dimensionality!");
133 
134  using Idx = idx::Idx<TExtent>;
135 
136  auto & view(this->m_view);
137  auto const & extent(this->m_extent);
138 
139  auto const extentWidth(extent::getWidth(extent));
140 
141  if(extentWidth == 0)
142  {
143  return;
144  }
145 
146  auto const extentWidthBytes(extentWidth * static_cast<Idx>(sizeof(elem::Elem<TView>)));
147 #if !defined(NDEBUG)
148  auto const dstWidth(extent::getWidth(view));
149 #endif
150  auto const dstNativePtr(reinterpret_cast<void *>(mem::view::getPtrNative(view)));
151  ALPAKA_ASSERT(extentWidth <= dstWidth);
152 
153  // Set the current device.
154  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
155  ALPAKA_API_PREFIX(SetDevice)(
156  this->m_iDevice));
157  // Initiate the memory set.
158  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
159  ALPAKA_API_PREFIX(MemsetAsync)(
160  dstNativePtr,
161  static_cast<int>(this->m_byte),
162  static_cast<size_t>(extentWidthBytes),
163  queue.m_spQueueImpl->m_UniformCudaHipQueue));
164  }
165  };

252  TView & view,

259  //-----------------------------------------------------------------------------
260  template<
261  typename TQueue
262  >
263  auto enqueue(TQueue & queue) const
264  -> void
265  {
266  static_assert(
267  dim::Dim<TView>::value == 3u,
268  "The destination buffer is required to be 3-dimensional for this specialization!");
269  static_assert(
270  dim::Dim<TView>::value == dim::Dim<TExtent>::value,
271  "The destination buffer and the extent are required to have the same dimensionality!");
272 
273  using Elem = alpaka::elem::Elem<TView>;
274  using Idx = idx::Idx<TExtent>;
275 
276  auto & view(this->m_view);
277  auto const & extent(this->m_extent);
278 
279  auto const extentWidth(extent::getWidth(extent));
280  auto const extentHeight(extent::getHeight(extent));
281  auto const extentDepth(extent::getDepth(extent));
282 
283  // This is not only an optimization but also prevents a division by zero.
284  if(extentWidth == 0 || extentHeight == 0 || extentDepth == 0)
285  {
286  return;
287  }
288 
289  auto const dstWidth(extent::getWidth(view));
290 #if !defined(NDEBUG)
291  auto const dstHeight(extent::getHeight(view));
292  auto const dstDepth(extent::getDepth(view));
293 #endif
294  auto const dstPitchBytesX(mem::view::getPitchBytes<dim::Dim<TView>::value - 1u>(view));
295  auto const dstPitchBytesY(mem::view::getPitchBytes<dim::Dim<TView>::value - (2u % dim::Dim<TView>::value)>(view));
296  auto const dstNativePtr(reinterpret_cast<void *>(mem::view::getPtrNative(view)));
297  ALPAKA_ASSERT(extentWidth <= dstWidth);
298  ALPAKA_ASSERT(extentHeight <= dstHeight);
299  ALPAKA_ASSERT(extentDepth <= dstDepth);
300 
301  // Fill CUDA parameter structures.
302  ALPAKA_API_PREFIX(PitchedPtr) const pitchedPtrVal(
303  ALPAKA_PP_CONCAT(make_,ALPAKA_API_PREFIX(PitchedPtr))(
304  dstNativePtr,
305  static_cast<size_t>(dstPitchBytesX),
306  static_cast<size_t>(dstWidth * static_cast<Idx>(sizeof(Elem))),
307  static_cast<size_t>(dstPitchBytesY / dstPitchBytesX)));
308 
309  ALPAKA_API_PREFIX(Extent) const extentVal(
310  ALPAKA_PP_CONCAT(make_,ALPAKA_API_PREFIX(Extent))(
311  static_cast<size_t>(extentWidth * static_cast<Idx>(sizeof(Elem))),
312  static_cast<size_t>(extentHeight),
313  static_cast<size_t>(extentDepth)));
314 
315  // Set the current device.
316  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
317  ALPAKA_API_PREFIX(SetDevice)(
318  this->m_iDevice));
319  // Initiate the memory set.
320  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
321  ALPAKA_API_PREFIX(Memset3DAsync)(
322  pitchedPtrVal,
323  static_cast<int>(this->m_byte),
324  extentVal,
325  queue.m_spQueueImpl->m_UniformCudaHipQueue));
326  }
327  };
