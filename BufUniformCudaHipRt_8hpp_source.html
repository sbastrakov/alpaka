65  //! The CUDA/HIP memory buffer.
66  template<
67  typename TElem,
68  typename TDim,
69  typename TIdx>
70  class BufUniformCudaHipRt
71  {
72  static_assert(
73  !std::is_const<TElem>::value,
74  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const elements!");
75  static_assert(
76  !std::is_const<TIdx>::value,
77  "The idx type of the buffer can not be const!");
78  private:
79  using Elem = TElem;
80  using Dim = TDim;
81 
82  public:
83  //-----------------------------------------------------------------------------
84  //! Constructor
85  template<
86  typename TExtent>
87  ALPAKA_FN_HOST BufUniformCudaHipRt(
88  dev::DevUniformCudaHipRt const & dev,
89  TElem * const pMem,
90  TIdx const & pitchBytes,
91  TExtent const & extent) :
92  m_dev(dev),
93  m_extentElements(extent::getExtentVecEnd<TDim>(extent)),
94  m_spMem(
95  pMem,
96  // NOTE: Because the BufUniformCudaHipRt object can be copied and the original object could have been destroyed,
97  // a std::ref(m_dev) or a this pointer can not be bound to the callback because they are not always valid at time of destruction.
98  std::bind(&BufUniformCudaHipRt::freeBuffer, std::placeholders::_1, m_dev)),
99  m_pitchBytes(pitchBytes)
100  {
101  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
102 
103  static_assert(
104  TDim::value == dim::Dim<TExtent>::value,
105  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be identical!");
106  static_assert(
107  std::is_same<TIdx, idx::Idx<TExtent>>::value,
108  "The idx type of TExtent and the TIdx template parameter have to be identical!");
109  }
110 
111  private:
112  //-----------------------------------------------------------------------------
113  //! Frees the shared buffer.
114  ALPAKA_FN_HOST static auto freeBuffer(
115  TElem * const memPtr,
116  dev::DevUniformCudaHipRt const & dev)
117  -> void
118  {
119  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
120 
121  // Set the current device.
122  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
123  ALPAKA_API_PREFIX(SetDevice)(
124  dev.m_iDevice));
125  // Free the buffer.
126  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
127  ALPAKA_API_PREFIX(Free)(reinterpret_cast<void *>(memPtr)));
128  }
129 
130  public:
131  dev::DevUniformCudaHipRt m_dev; // NOTE: The device has to be destructed after the memory pointer because it is required for destruction.
132  vec::Vec<TDim, TIdx> m_extentElements;
133  std::shared_ptr<TElem> m_spMem;
134  TIdx m_pitchBytes;
135  };
136  }

333  TElem,

335  TIdx,

338  //-----------------------------------------------------------------------------
339  template<
340  typename TExtent>

342  dev::DevUniformCudaHipRt const & dev,
343  TExtent const & extent)
344  -> mem::buf::BufUniformCudaHipRt<TElem, dim::DimInt<1u>, TIdx>
345  {
346  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
347 
348  auto const width(extent::getWidth(extent));
349  auto const widthBytes(width * static_cast<TIdx>(sizeof(TElem)));
350 
351  // Set the current device.
352  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
353  ALPAKA_API_PREFIX(SetDevice)(
354  dev.m_iDevice));
355  // Allocate the buffer on this device.
356  void * memPtr;
357  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
358  ALPAKA_API_PREFIX(Malloc)(
359  &memPtr,
360  static_cast<std::size_t>(widthBytes)));
361 
362 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
363  std::cout << __func__
364  << " ew: " << width
365  << " ewb: " << widthBytes
366  << " ptr: " << memPtr
367  << std::endl;
368 #endif
369  return
370  mem::buf::BufUniformCudaHipRt<TElem, dim::DimInt<1u>, TIdx>(
371  dev,
372  reinterpret_cast<TElem *>(memPtr),
373  static_cast<TIdx>(widthBytes),
374  extent);
375  }
376  };

413  ALPAKA_API_PREFIX(SetDevice)(

415 
416 
417  // Allocate the buffer on this device.
418  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
419  ALPAKA_API_PREFIX(MallocPitch)(
420  &memPtr,
421  &pitchBytes,
422  static_cast<std::size_t>(widthBytes),
423  static_cast<std::size_t>(height)));
424  ALPAKA_ASSERT(pitchBytes >= static_cast<std::size_t>(widthBytes) || (width * height) == 0);
425  }
426 
427 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
428  std::cout << __func__
429  << " ew: " << width
430  << " eh: " << height
431  << " ewb: " << widthBytes
432  << " ptr: " << memPtr
433  << " pitch: " << pitchBytes
434  << std::endl;
435 #endif
436  return
437  mem::buf::BufUniformCudaHipRt<TElem, dim::DimInt<2u>, TIdx>(
438  dev,
439  reinterpret_cast<TElem *>(memPtr),
440  static_cast<TIdx>(pitchBytes),
441  extent);
442  }

697  template<
698  typename TElem,
699  typename TDim,
700  typename TIdx>
701  struct Map<
702  mem::buf::BufCpu<TElem, TDim, TIdx>,
703  dev::DevUniformCudaHipRt>
704  {
705  //-----------------------------------------------------------------------------
706  ALPAKA_FN_HOST static auto map(
707  mem::buf::BufCpu<TElem, TDim, TIdx> & buf,
708  dev::DevUniformCudaHipRt const & dev)
709  -> void
710  {
711  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
712 
713  // If it is already the same device, nothing has to be mapped.
714  if(dev::getDev(buf) != dev)
715  {
716  // cuda/hip-HostRegisterMapped:
717  // Maps the allocation into the CUDA/HIP address space.The device pointer to the memory may be obtained by calling cudaHostGetDevicePointer().
718  // This feature is available only on GPUs with compute capability greater than or equal to 1.1.
