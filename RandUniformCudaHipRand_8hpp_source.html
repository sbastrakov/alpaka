178  template<>
179  class UniformReal<
180  float>
181  {
182  public:
183  //-----------------------------------------------------------------------------
184  UniformReal() = default;
185 
186  //-----------------------------------------------------------------------------
187  template<
188  typename TGenerator>
189  __device__ auto operator()(
190  TGenerator & generator)
191  -> float
192  {
193  // (0.f, 1.0f]
194 #ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
195  float const fUniformRand(curand_uniform(&generator.m_State));
196 #else
197  float const fUniformRand(hiprand_uniform(&generator.m_State));
198 #endif
199  // NOTE: (1.0f - curand_uniform) does not work, because curand_uniform seems to return denormalized floats around 0.f.
200  // [0.f, 1.0f)
201  return fUniformRand * static_cast<float>( fUniformRand != 1.0f );
202  }
203  };
