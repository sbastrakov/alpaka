40  //! The CPU device memory copy task base.
41  //!
42  //! Copies from CPU memory into CPU memory.
43  template<
44  typename TDim,
45  typename TViewDst,
46  typename TViewSrc,
47  typename TExtent>
48  struct TaskCopyCpuBase
49  {
50  using ExtentSize = idx::Idx<TExtent>;
51  using DstSize = idx::Idx<TViewDst>;
52  using SrcSize = idx::Idx<TViewSrc>;
53  using Elem = elem::Elem<TViewSrc>;
54 
55  static_assert(
56  !std::is_const<TViewDst>::value,
57  "The destination view can not be const!");
58 
59  static_assert(
60  dim::Dim<TViewDst>::value == dim::Dim<TViewSrc>::value,
61  "The source and the destination view are required to have the same dimensionality!");
62  static_assert(
63  dim::Dim<TViewDst>::value == dim::Dim<TExtent>::value,
64  "The views and the extent are required to have the same dimensionality!");
65  static_assert(
66  dim::Dim<TViewDst>::value == TDim::value,
67  "The destination view and the input TDim are required to have the same dimensionality!");
68 
69  static_assert(
70  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
71  "The destination view and the extent are required to have compatible idx type!");
72  static_assert(
73  meta::IsIntegralSuperset<SrcSize, ExtentSize>::value,
74  "The source view and the extent are required to have compatible idx type!");
75 
76  static_assert(
77  std::is_same<elem::Elem<TViewDst>, std::remove_const_t<elem::Elem<TViewSrc>>>::value,
78  "The source and the destination view are required to have the same element type!");
79 
80  //-----------------------------------------------------------------------------
81  TaskCopyCpuBase(
82  TViewDst & viewDst,
83  TViewSrc const & viewSrc,
84  TExtent const & extent) :
85  m_extent(extent::getExtentVec(extent)),
86  m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem))),
87 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
88  m_dstExtent(extent::getExtentVec(viewDst)),
89  m_srcExtent(extent::getExtentVec(viewSrc)),
90 #endif
91  m_dstPitchBytes(mem::view::getPitchBytesVec(viewDst)),
92  m_srcPitchBytes(mem::view::getPitchBytesVec(viewSrc)),
93 
94  m_dstMemNative(reinterpret_cast<std::uint8_t *>(mem::view::getPtrNative(viewDst))),
95  m_srcMemNative(reinterpret_cast<std::uint8_t const *>(mem::view::getPtrNative(viewSrc)))
96  {
97  ALPAKA_ASSERT((vec::cast<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
98  ALPAKA_ASSERT((vec::cast<SrcSize>(m_extent) <= m_srcExtent).foldrAll(std::logical_or<bool>()));
99  ALPAKA_ASSERT(static_cast<DstSize>(m_extentWidthBytes) <= m_dstPitchBytes[TDim::value - 1u]);
100  ALPAKA_ASSERT(static_cast<SrcSize>(m_extentWidthBytes) <= m_srcPitchBytes[TDim::value - 1u]);
101  }
102 
103 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
104  //-----------------------------------------------------------------------------
105  ALPAKA_FN_HOST auto printDebug() const
106  -> void
107  {
108  std::cout << __func__
109  << " e: " << m_extent

115  << " sptr: " << reinterpret_cast<void const *>(m_srcMemNative)
116  << " spitchb: " << m_srcPitchBytes
117  << std::endl;
118  }
119 #endif

137  //! The CPU device ND memory copy task.
138  template<
139  typename TDim,
140  typename TViewDst,
141  typename TViewSrc,
142  typename TExtent>
143  struct TaskCopyCpu : public TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>
144  {
145  using DimMin1 = dim::DimInt<TDim::value - 1u>;
146  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
147  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
148  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
149 
150  //-----------------------------------------------------------------------------
151  using TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyCpuBase;
152 
153  //-----------------------------------------------------------------------------
154  ALPAKA_FN_HOST auto operator()() const
155  -> void
156  {
157  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
158 
159 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
160  this->printDebug();
161 #endif
162  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
163  vec::Vec<DimMin1, ExtentSize> const extentWithoutInnermost(vec::subVecBegin<DimMin1>(this->m_extent));
164  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
165  vec::Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(vec::subVecEnd<DimMin1>(this->m_dstPitchBytes));
166  vec::Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(vec::subVecEnd<DimMin1>(this->m_srcPitchBytes));
167 
168  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
169  {
170  meta::ndLoopIncIdx(
171  extentWithoutInnermost,
172  [&](vec::Vec<DimMin1, ExtentSize> const & idx)
173  {
174  std::memcpy(
175  reinterpret_cast<void *>(this->m_dstMemNative + (vec::cast<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
176  reinterpret_cast<void const *>(this->m_srcMemNative + (vec::cast<SrcSize>(idx) * srcPitchBytesWithoutOutmost).foldrAll(std::plus<SrcSize>())),
177  static_cast<std::size_t>(this->m_extentWidthBytes));
178  });
179  }
180  }
181  };
