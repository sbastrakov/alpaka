178  //! The GPU CUDA/HIP accelerator atomic operation.
179  template<
180  typename THierarchy>
181  struct AtomicOp<
182  op::Add,
183  atomic::AtomicUniformCudaHipBuiltIn,
184  double,
185  THierarchy>
186  {
187  //-----------------------------------------------------------------------------
188  __device__ static auto atomicOp(
189  atomic::AtomicUniformCudaHipBuiltIn const &,
190  double * const addr,
191  double const & value)
192  -> double
193  {
194 #if BOOST_ARCH_PTX >= BOOST_VERSION_NUMBER(6, 0, 0)
195  return atomicAdd(addr, value);
196 #else
197  // Code from: http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
198 
199  unsigned long long int * address_as_ull(reinterpret_cast<unsigned long long int *>(addr));
200  unsigned long long int old(*address_as_ull);
201  unsigned long long int assumed;
202  do
203  {
204  assumed = old;
205  old = atomicCAS(
206  address_as_ull,
207  assumed,
208  static_cast<unsigned long long>(__double_as_longlong(value + __longlong_as_double(static_cast<long long>(assumed)))));
209  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
210  }
211  while(assumed != old);
212  return __longlong_as_double(static_cast<long long>(old));
213 #endif
214  }
215  };
