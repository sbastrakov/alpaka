61  ///! copy 3D memory
62  ///
63  /// It is required to start `height * depth` HIP/CUDA blocks.
64  /// The kernel loops over the memory rows.
65  template<typename T>
66  __global__ void hipMemcpy3DEmulatedKernelD2D(
67  char * dstPtr, vec2D const dstPitch,
68  char const * srcPtr, vec2D const srcPitch, vec3D const extent
69  )
70  {
71  constexpr size_t X = 2;
72  constexpr size_t Y = 1;
73  constexpr size_t Z = 0;
74 
75  // blockDim.[y,z] is always 1 and is not needed for index calculations
76  // gridDim and blockIdx is already in alpaka index order [z,y,x]
77 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
78  alpaka::vec::Vec<alpaka::dim::DimInt<3u>, uint32_t> const tid(
79  blockIdx.x,
80  blockIdx.y,
81  blockIdx.z * blockDim.x + threadIdx.x);
82 
83  size_t const bytePerBlock = sizeof(T) * blockDim.x;
84  size_t const bytePerRow = gridDim.z * bytePerBlock;
85 #else
86  alpaka::vec::Vec<alpaka::dim::DimInt<3u>, uint32_t> const tid(
87  hipBlockIdx_x,
88  hipBlockIdx_y,
89  hipBlockIdx_z * hipBlockDim_x + hipThreadIdx_x);
90 
91  size_t const bytePerBlock = sizeof(T) * hipBlockDim_x;
92  size_t const bytePerRow = hipGridDim_z * bytePerBlock;
93 #endif
94 
95  size_t const peelLoopSteps = extent[X] / bytePerRow;
96  bool const needRemainder = (extent[X] % bytePerRow) != 0;
97 
98  dstPtr += tid[Z] * dstPitch[Z] + tid[Y] * dstPitch[Y];
99  srcPtr += tid[Z] * srcPitch[Z] + tid[Y] * srcPitch[Y];
100 
101  for(size_t idx = 0; idx < peelLoopSteps; ++idx)
102  {
103  size_t const byteOffsetX = idx * bytePerRow + tid[X] * sizeof(T);
104  auto dst = reinterpret_cast<T *>(dstPtr + byteOffsetX);
105  auto src = reinterpret_cast<T const *>(srcPtr + byteOffsetX);
106  *dst = *src;
107  }
108  if(needRemainder)
109  {
110  size_t const byteOffsetX = peelLoopSteps * bytePerRow + tid[X] * sizeof(T);
111  if(byteOffsetX < extent[X])
112  {
113  auto dst = reinterpret_cast<T *>(dstPtr + byteOffsetX);
114  auto src = reinterpret_cast<T const *>(srcPtr + byteOffsetX);
115  *dst = *src;
116  }
117  }
118  }

120  inline size_t divUp(size_t const & x, size_t const & y)
121  {
122  return (x + y - 1u) / y;
123  }

125  inline auto memcpy3DEmulatedD2DAsync(ALPAKA_API_PREFIX(Memcpy3DParms) const * const p, ALPAKA_API_PREFIX(Stream_t) stream)
126  {
127  using dim3Value_t = std::remove_reference_t<decltype(std::declval<dim3>().x)>;
128  // extent[2] is in byte
129  vec3D const extent(p->extent.depth, p->extent.height, p->extent.width);
130  // pitch in bytes
131  vec2D const dstPitch(p->dstPtr.pitch * p->dstPtr.ysize,p->dstPtr.pitch);
132  vec2D const srcPitch(p->srcPtr.pitch * p->srcPtr.ysize,p->srcPtr.pitch);
133  // offset[2] is in byte
134  vec3D const dstOffset(p->dstPos.z,p->dstPos.y,p->dstPos.x);
135  vec3D const srcOffset(p->srcPos.z,p->srcPos.y,p->srcPos.x);
136 
137  char const * srcPtr =
138  reinterpret_cast<char const *>(p->srcPtr.ptr) + srcOffset[0] * srcPitch[0] + srcOffset[1] * srcPitch[1] + srcOffset[2];
139  char * dstPtr =
140  reinterpret_cast<char *>(p->dstPtr.ptr) + dstOffset[0] * dstPitch[0] + dstOffset[1] * dstPitch[1] + dstOffset[2];
141 
142  bool const use4Byte = (reinterpret_cast<uintptr_t>(srcPtr) % 4u) == 0u && (reinterpret_cast<uintptr_t>(dstPtr) % 4u) == 0u && (extent[2] % 4u) == 0u;
143 
144  if(use4Byte)
145  {
146  dim3 block(static_cast<dim3Value_t>(std::min(divUp(extent[2], 4u), size_t(256u))));
147  // use alpaka index order [z,y,x] because x is by default 1
148  dim3 grid(static_cast<dim3Value_t>(extent[0]), static_cast<dim3Value_t>(extent[1]), 1u);
149  // for less than 100 blocks increase the number of blocks used to copy a row to
150  // increase the utilization of the device
151  if(grid.x * grid.y < 100)
152  {
153  grid.z = static_cast<dim3Value_t>(divUp(divUp(extent[2], 4u), block.x));
154  }
155 
156 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
157  hipMemcpy3DEmulatedKernelD2D<int><<<
158  grid, block, 0, stream>>>(
159  dstPtr, dstPitch, srcPtr, srcPitch, extent);
160 #else
161  hipLaunchKernelGGL(
162  HIP_KERNEL_NAME(hipMemcpy3DEmulatedKernelD2D<int>),
163  grid, block, 0, stream,
164  dstPtr, dstPitch, srcPtr, srcPitch, extent);
165 #endif
166  }
167  else
168  {
169  dim3 block(static_cast<dim3Value_t>(std::min(extent[2], size_t(256u))));
170  // use alpaka index order [z,y,x] because x is by default 1
171  dim3 grid(static_cast<dim3Value_t>(extent[0]), static_cast<dim3Value_t>(extent[1]), 1u);
172  // for less than 100 blocks increase the number of blocks used to copy a row to
173  // increase the utilization of the device
174  if(grid.x * grid.y < 100)
175  {
176  grid.z = static_cast<dim3Value_t>(divUp(extent[2], block.x));
177  }
178 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
179  hipMemcpy3DEmulatedKernelD2D<char><<<
180  grid, block, 0, stream>>>(
181  dstPtr, dstPitch, srcPtr, srcPitch, extent);
182 #else
183  hipLaunchKernelGGL(
184  HIP_KERNEL_NAME(hipMemcpy3DEmulatedKernelD2D<char>),
185  grid, block, 0, stream,
186  dstPtr, dstPitch, srcPtr, srcPitch, extent);
187 #endif
188  }
189  return ALPAKA_API_PREFIX(GetLastError)();
190  };

193  //! Not being able to enable peer access does not prevent such device to device memory copies.
194  //! However, those copies may be slower because the memory is copied via the CPU.
195  inline auto enablePeerAccessIfPossible(
196  const int & devSrc,
197  const int & devDst)
198  -> void
199  {
200  ALPAKA_ASSERT(devSrc != devDst);
201 
202 #if BOOST_COMP_CLANG
203  #pragma clang diagnostic push
204  #pragma clang diagnostic ignored "-Wexit-time-destructors"
205 #endif
206  static std::set<std::pair<int, int>> alreadyCheckedPeerAccessDevices;
207 #if BOOST_COMP_CLANG
208  #pragma clang diagnostic pop
209 #endif
210  auto const devicePair = std::make_pair(devSrc, devDst);
211 
212  if(alreadyCheckedPeerAccessDevices.find(devicePair) == alreadyCheckedPeerAccessDevices.end())
213  {
214  alreadyCheckedPeerAccessDevices.insert(devicePair);
215 
216  int canAccessPeer = 0;
217  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(DeviceCanAccessPeer)(&canAccessPeer, devSrc, devDst));
218 
219  if(!canAccessPeer) {
220 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
221  std::cout << __func__
222  << " Direct peer access between given GPUs is not possible!"
223  << " src=" << devSrc
224  << " dst=" << devDst
225  << std::endl;
226 #endif
227  return;
228  }
229  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(devSrc));
230 
231  // NOTE: "until access is explicitly disabled using cudaDeviceDisablePeerAccess() or either device is reset using cudaDeviceReset()."
232  // We do not remove a device from the enabled device pairs on cudaDeviceReset.
233  // Note that access granted by this call is unidirectional and that in order to access memory on the current device from peerDevice, a separate symmetric call to cudaDeviceEnablePeerAccess() is required.
234  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(DeviceEnablePeerAccess)(devDst, 0));
235  }
236  }

673  {

675  printDebug();
676 #endif
677  // This is not only an optimization but also prevents a division by zero.
678  if(m_extentWidthBytes == 0 || m_extentHeight == 0 || m_extentDepth == 0)
679  {
680  return;
681  }
682 
683  if(m_iDstDevice == m_iSrcDevice)
684  {
685  // Create the struct describing the copy.
686  ALPAKA_API_PREFIX(Memcpy3DParms) const uniformCudaHipMemCpy3DParms(
687  buildUniformCudaHipMemcpy3DParms());
688  // Set the current device.
689  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
690  ALPAKA_API_PREFIX(SetDevice)(
691  m_iDstDevice));
692 #if defined(ALPAKA_EMU_MEMCPY3D_ENABLED)
693  auto isDevice2DeviceCopy = m_uniformMemCpyKind == ALPAKA_API_PREFIX(MemcpyDeviceToDevice);
694  // contiguous memory can be copied by a single 1D memory copy
695  auto isContiguousMemory = uniformCudaHipMemCpy3DParms.extent.width == uniformCudaHipMemCpy3DParms.dstPtr.pitch &&
696  uniformCudaHipMemCpy3DParms.extent.width == uniformCudaHipMemCpy3DParms.srcPtr.pitch;
697  if(isDevice2DeviceCopy && !isContiguousMemory)
698  {
699  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
700  uniform_cuda_hip::detail::memcpy3DEmulatedD2DAsync(
701  &uniformCudaHipMemCpy3DParms,
702  queue.m_spQueueImpl->m_UniformCudaHipQueue));
703  }
704  else
705 #endif
706  {
707  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
708  ALPAKA_API_PREFIX(Memcpy3DAsync)(
709  &uniformCudaHipMemCpy3DParms,
710  queue.m_spQueueImpl->m_UniformCudaHipQueue));
711  }
712  }
713  else
714  {
715  alpaka::mem::view::uniform_cuda_hip::detail::enablePeerAccessIfPossible(m_iSrcDevice, m_iDstDevice);
