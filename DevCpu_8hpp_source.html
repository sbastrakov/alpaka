83  //-----------------------------------------------------------------------------
84  ALPAKA_FN_HOST auto getAllExistingQueues() const
85  -> std::vector<std::shared_ptr<queue::cpu::ICpuQueue>>
86  {
87  std::vector<std::shared_ptr<queue::cpu::ICpuQueue>> vspQueues;
88 
89  std::lock_guard<std::mutex> lk(m_Mutex);
90  vspQueues.reserve(m_queues.size());
91 
92  for(auto it = m_queues.begin(); it != m_queues.end();)
93  {
94  auto spQueue(it->lock());
95  if(spQueue)
96  {
97  vspQueues.emplace_back(std::move(spQueue));
98  ++it;
99  }
100  else
101  {
102  it = m_queues.erase(it);
103  }
104  }
105  return vspQueues;
106  }

112  -> void
113  {
114  std::lock_guard<std::mutex> lk(m_Mutex);
115 
116  // Register this queue on the device.
117  m_queues.push_back(spQueue);
118  }
119

121  std::mutex mutable m_Mutex;
122  std::vector<std::weak_ptr<queue::cpu::ICpuQueue>> mutable m_queues;
