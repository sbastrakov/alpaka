46  //! The CPU memory buffer.
47  template<
48  typename TElem,
49  typename TDim,
50  typename TIdx>
51  class BufCpuImpl final :
52  public mem::alloc::AllocCpuBoostAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>
53  {
54  static_assert(
55  !std::is_const<TElem>::value,
56  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const elements!");
57  static_assert(
58  !std::is_const<TIdx>::value,
59  "The idx type of the buffer can not be const!");
60  public:
61  //-----------------------------------------------------------------------------
62  template<
63  typename TExtent>
64  ALPAKA_FN_HOST BufCpuImpl(
65  dev::DevCpu const & dev,
66  TExtent const & extent) :
67  mem::alloc::AllocCpuBoostAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>(),
68  m_dev(dev),
69  m_extentElements(extent::getExtentVecEnd<TDim>(extent)),
70  m_pMem(mem::alloc::alloc<TElem>(*this, static_cast<std::size_t>(computeElementCount(extent)))),
71  m_pitchBytes(static_cast<TIdx>(extent::getWidth(extent) * static_cast<TIdx>(sizeof(TElem))))
72 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
73  ,m_bPinned(false)
74 #endif
75  {
76  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
77 
78  static_assert(
79  TDim::value == dim::Dim<TExtent>::value,
80  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be identical!");
81  static_assert(
82  std::is_same<TIdx, idx::Idx<TExtent>>::value,
83  "The idx type of TExtent and the TIdx template parameter have to be identical!");
84 
85 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
86  std::cout << __func__
87  << " e: " << m_extentElements
88  << " ptr: " << static_cast<void *>(m_pMem)
89  << " pitch: " << m_pitchBytes
90  << std::endl;
91 #endif
92  }
93  //-----------------------------------------------------------------------------
94  BufCpuImpl(BufCpuImpl const &) = delete;
95  //-----------------------------------------------------------------------------
96  BufCpuImpl(BufCpuImpl &&) = default;
97  //-----------------------------------------------------------------------------
98  auto operator=(BufCpuImpl const &) -> BufCpuImpl & = delete;
99  //-----------------------------------------------------------------------------
100  auto operator=(BufCpuImpl &&) -> BufCpuImpl & = default;
101  //-----------------------------------------------------------------------------
102  ALPAKA_FN_HOST ~BufCpuImpl()
103  {
104  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
105 
106 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
107  // Unpin this memory if it is currently pinned.
108  mem::buf::unpin(*this);
109 #endif
110  // NOTE: m_pMem is allowed to be a nullptr here.
111  mem::alloc::free(*this, m_pMem);
112  }
113 
114  private:
115  //-----------------------------------------------------------------------------
116  //! \return The number of elements to allocate.
117  template<
118  typename TExtent>
119  ALPAKA_FN_HOST static auto computeElementCount(
120  TExtent const & extent)
121  -> TIdx
122  {
123  auto const extentElementCount(extent::getExtentProduct(extent));
124 
125  return extentElementCount;
126  }
127 
128  public:
129  dev::DevCpu const m_dev;
130  vec::Vec<TDim, TIdx> const m_extentElements;
131  TElem * const m_pMem;
132  TIdx const m_pitchBytes;
133 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
134  bool m_bPinned;
135 #endif
136  };
137  }

442  //! The BufCpu memory pinning trait specialization.
443  template<
444  typename TElem,
445  typename TDim,
446  typename TIdx>
447  struct Pin<
448  mem::buf::BufCpu<TElem, TDim, TIdx>>
449  {
450  //-----------------------------------------------------------------------------
451  ALPAKA_FN_HOST static auto pin(
452  mem::buf::BufCpu<TElem, TDim, TIdx> & buf)
453  -> void
454  {
455  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
456 
457  if(!mem::buf::isPinned(buf))
458  {
459 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
460  if(buf.m_spBufCpuImpl->m_extentElements.prod() != 0)
461  {
462  // - cudaHostRegisterDefault:
463  // See http://cgi.cs.indiana.edu/~nhusted/dokuwiki/doku.php?id=programming:cudaperformance1
464  // - cudaHostRegisterPortable:
465  // The memory returned by this call will be considered as pinned memory by all CUDA contexts, not just the one that performed the allocation.
466  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_IGNORE(
467  ALPAKA_API_PREFIX(HostRegister)(
468  const_cast<void *>(reinterpret_cast<void const *>(mem::view::getPtrNative(buf))),
469  extent::getExtentProduct(buf) * sizeof(elem::Elem<buf::BufCpu<TElem, TDim, TIdx>>),
470  ALPAKA_API_PREFIX(HostRegisterDefault)),
471  ALPAKA_API_PREFIX(ErrorHostMemoryAlreadyRegistered));
472 
473  buf.m_spBufCpuImpl->m_bPinned = true;
474  }
475 #else
476  static_assert(
477  meta::DependentFalseType<TElem>::value,
478  "Memory pinning of BufCpu is not implemented when CUDA or HIP is not enabled!");
479 #endif
480  }
481  }
482  };
