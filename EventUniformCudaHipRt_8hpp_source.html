52  //! The CUDA/HIP RT device event implementation.
53  class EventUniformCudaHipImpl final
54  {
55  public:
56  //-----------------------------------------------------------------------------
57  ALPAKA_FN_HOST EventUniformCudaHipImpl(
58  dev::DevUniformCudaHipRt const & dev,
59  bool bBusyWait) :
60  m_dev(dev),
61  m_UniformCudaHipEvent()
62  {
63  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
64 
65  // Set the current device.
66  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
67  ALPAKA_API_PREFIX(SetDevice)(
68  m_dev.m_iDevice));
69 
70  // Create the event on the current device with the specified flags. Valid flags include:
71  // - cuda/hip-EventDefault: Default event creation flag.
72  // - cuda/hip-EventBlockingSync : Specifies that event should use blocking synchronization.
73  // A host thread that uses cuda/hip-EventSynchronize() to wait on an event created with this flag will block until the event actually completes.
74  // - cuda/hip-EventDisableTiming : Specifies that the created event does not need to record timing data.
75  // Events created with this flag specified and the cuda/hip-EventBlockingSync flag not specified will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
76  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
77  ALPAKA_API_PREFIX(EventCreateWithFlags)(
78  &m_UniformCudaHipEvent,
79  (bBusyWait ? ALPAKA_API_PREFIX(EventDefault) : ALPAKA_API_PREFIX(EventBlockingSync)) | ALPAKA_API_PREFIX(EventDisableTiming)));
80  }
81  //-----------------------------------------------------------------------------
82  EventUniformCudaHipImpl(EventUniformCudaHipImpl const &) = delete;
83  //-----------------------------------------------------------------------------
84  EventUniformCudaHipImpl(EventUniformCudaHipImpl &&) = default;
85  //-----------------------------------------------------------------------------
86  auto operator=(EventUniformCudaHipImpl const &) -> EventUniformCudaHipImpl & = delete;
87  //-----------------------------------------------------------------------------
88  auto operator=(EventUniformCudaHipImpl &&) -> EventUniformCudaHipImpl & = delete;
89  //-----------------------------------------------------------------------------
90  ALPAKA_FN_HOST ~EventUniformCudaHipImpl()
91  {
92  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
93 
94  // Set the current device. \TODO: Is setting the current device before cuda/hip-EventDestroy required?
95 
96  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(
97  m_dev.m_iDevice));
98  // In case event has been recorded but has not yet been completed when cuda/hip-EventDestroy() is called, the function will return immediately
99  // and the resources associated with event will be released automatically once the device has completed event.
100  // -> No need to synchronize here.
101  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(EventDestroy)(
102  m_UniformCudaHipEvent));
103  }
104 
105  public:
106  dev::DevUniformCudaHipRt const m_dev; //!< The device this event is bound to.
107 
108  ALPAKA_API_PREFIX(Event_t) m_UniformCudaHipEvent;
109  };
110  }
