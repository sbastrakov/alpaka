53  template <typename T>
54  void alloc() const
55  {
56  m_allocdBytes = allocPitch<T>();
57  uint8_t* buf = &m_mem[m_allocdBytes];
58  new (buf) T();
59  m_allocdBytes += sizeof(T);
60 #if (defined ALPAKA_DEBUG_OFFLOAD_ASSUME_HOST) && (! defined NDEBUG)
61  ALPAKA_ASSERT(m_allocdBytes < m_capacity);
62 #endif
63  }

90  template<typename T>
91  unsigned int allocPitch() const
92  {
93  static_assert(
94  core::vectorization::defaultAlignment >= alignof(T),
95  "Unable to get block shared static memory for types with alignment higher than defaultAlignment!");
96  constexpr unsigned int align = std::max(TDataAlignBytes, static_cast<unsigned int>(alignof(T)));
97  return (m_allocdBytes/align + (m_allocdBytes%align>0))*align;
98  }
