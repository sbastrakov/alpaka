311  public:

314  mem::view::ViewSubView<TDev, TElem, TDim, TIdx> const & view)

324  ALPAKA_FN_HOST static auto getPtrNative(

327  {
328  // \TODO: pre-calculate this pointer for faster execution.
329  return
330  reinterpret_cast<TElem *>(
331  reinterpret_cast<std::uint8_t *>(mem::view::getPtrNative(view.m_viewParentView))
332  + pitchedOffsetBytes(view, IdxSequence()));
333  }
334 
335  private:
336  //-----------------------------------------------------------------------------
337  //! For a 3D vector this calculates:
338  //!
339  //! offset::getOffset<0u>(view) * mem::view::getPitchBytes<1u>(view)
340  //! + offset::getOffset<1u>(view) * mem::view::getPitchBytes<2u>(view)
341  //! + offset::getOffset<2u>(view) * mem::view::getPitchBytes<3u>(view)
342  //! while mem::view::getPitchBytes<3u>(view) is equivalent to sizeof(TElem)
343  template<
344  typename TView,
345  std::size_t... TIndices>
346  ALPAKA_FN_HOST static auto pitchedOffsetBytes(
347  TView const & view,
348  std::index_sequence<TIndices...> const &)
349  -> TIdx
350  {
351  return
352  meta::foldr(
353  std::plus<TIdx>(),
354  pitchedOffsetBytesDim<TIndices>(view)...);
355  }
356  //-----------------------------------------------------------------------------
357  template<
358  std::size_t Tidx,
359  typename TView>
360  ALPAKA_FN_HOST static auto pitchedOffsetBytesDim(
361  TView const & view)
362  -> TIdx
363  {
364  return
365  offset::getOffset<Tidx>(view)
366  * mem::view::getPitchBytes<Tidx + 1u>(view);
367  }
368  };
