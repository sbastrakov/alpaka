133  //! The CUDA/HIP RT sync queue enqueue trait specialization.
134  template<
135  typename TTask>
136  struct Enqueue<
137  queue::QueueUniformCudaHipRtNonBlocking,
138  TTask>
139  {
140  //#############################################################################
141  enum class CallbackState
142  {
143  enqueued,
144  notified,
145  finished,
146  };
147 
148  //#############################################################################
149  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
150  {
151  std::mutex m_mutex;
152  std::condition_variable m_event;
153  CallbackState state = CallbackState::enqueued;
154  };
155 
156  //-----------------------------------------------------------------------------
157 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
158  static void CUDART_CB
159 #else
160  static void HIPRT_CB
161 #endif
162  uniformCudaHipRtCallback(ALPAKA_API_PREFIX(Stream_t) /*queue*/, ALPAKA_API_PREFIX(Error_t) /*status*/, void *arg)
163  {
164  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread has already finished.
165  const auto pCallbackSynchronizationData = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
166 
167  // Notify the executing thread.
168  {
169  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
170  pCallbackSynchronizationData->state = CallbackState::notified;
171  }
172  pCallbackSynchronizationData->m_event.notify_one();
173 
174  // Wait for the executing thread to finish the task if it has not already finished.
175  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
176  if(pCallbackSynchronizationData->state != CallbackState::finished)
177  {
178  pCallbackSynchronizationData->m_event.wait(
179  lock,
180  [pCallbackSynchronizationData](){
181  return pCallbackSynchronizationData->state == CallbackState::finished;
182  }
183  );
184  }
185  }
186 
187  //-----------------------------------------------------------------------------
188  ALPAKA_FN_HOST static auto enqueue(
189  queue::QueueUniformCudaHipRtNonBlocking & queue,
190  TTask const & task)
191  -> void
192  {
193 #if BOOST_COMP_HIP
194  // NOTE: hip callbacks are not blocking the stream.
195  // @todo remove this assert when hipStreamAddCallback is fixed
196  static_assert(
197  meta::DependentFalseType<TTask>::value,
198  "Callbacks are not supported for HIP-clang");
199 #endif
200 
201  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
202  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamAddCallback)(
203  queue.m_spQueueImpl->m_UniformCudaHipQueue,
204  uniformCudaHipRtCallback,
205  pCallbackSynchronizationData.get(),
206  0u));
207 
208  // We start a new std::thread which stores the task to be executed.
209  // This circumvents the limitation that it is not possible to call CUDA methods within the CUDA/HIP callback thread.
210  // The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
211  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
212  // before it executes the next task in the queue (CUDA/HIP stream).
213  std::thread t(
214  [pCallbackSynchronizationData, task](){
215  // If the callback has not yet been called, we wait for it.
216  {
217  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
218  if(pCallbackSynchronizationData->state != CallbackState::notified)
219  {
220  pCallbackSynchronizationData->m_event.wait(
221  lock,
222  [pCallbackSynchronizationData](){
223  return pCallbackSynchronizationData->state == CallbackState::notified;
224  }
225  );
226  }
227 
228  task();
229 
230  // Notify the waiting CUDA thread.
231  pCallbackSynchronizationData->state = CallbackState::finished;
232  }
233  pCallbackSynchronizationData->m_event.notify_one();
234  }
235  );
236 
237  t.detach();
238  }
239  };
240  }
241  }
242 }
